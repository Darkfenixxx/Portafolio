Código Ejemplo

if (PIND & (1 << INT_PIN)) {
		
		motor_estado = 0;
		PORTB &= ~(1 << MOTOR_PIN);
		} else {
		
		motor_estado = 1;
		PORTB |= (1 << MOTOR_PIN);
	}
